function copyText(){
var t=document.getElementById(‘et’).textContent;
if(navigator.clipboard&&navigator.clipboard.writeText){
navigator.clipboard.writeText(t).catch(function(){fallbackCopy(t);});
}else{fallbackCopy(t);}
var b=document.querySelector(’.copy-btn’);b.textContent=‘✅ 已复制！’;
setTimeout(function(){b.textContent=‘复制全部’;},2000);
}
function fallbackCopy(t){
var ta=document.createElement(‘textarea’);ta.value=t;ta.style.position=‘fixed’;ta.style.opacity=‘0’;
document.body.appendChild(ta);ta.focus();ta.select();
try{document.execCommand(‘copy’);}catch(e){}
document.body.removeChild(ta);
}
